#include <iostream>

using namespace std;

int main()
{
    int i=0,key=0,j=0;
    int array[10];
    //taking array input
    for(i=0;i<10;i++)
        cin>>array[i];
        
    
    for(i=2;i<10;i++)
    {   
        j=i-1;
        key=array[i];
        while(j>=0 && key<array[j])
        {
            array[j+1]=array[j];

        }
        array[j+1]=key;
    }
      for(i=0;i<10;i++)

}
